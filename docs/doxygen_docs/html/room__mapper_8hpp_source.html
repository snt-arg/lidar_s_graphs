60  inline float point_difference(int plane_type, pcl::PointXY p1, pcl::PointXY p2) {
61  float point_diff = 0;
62 
63  if(plane_type == PlaneUtils::plane_class::X_VERT_PLANE) {
64  p1.x = 0;
65  p2.x = 0;
66  point_diff = pcl::euclideanDistance(p1, p2);
67  }
68  if(plane_type == PlaneUtils::plane_class::Y_VERT_PLANE) {
69  p1.y = 0;
70  p2.y = 0;
71  point_diff = pcl::euclideanDistance(p1, p2);
72  }
73 
74  return point_diff;
75  }

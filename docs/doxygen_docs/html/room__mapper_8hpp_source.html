75 inline float point_difference(int plane_type, pcl::PointXY p1, pcl::PointXY p2) {
76 float point_diff = 0;
77
78 if(plane_type == PlaneUtils::plane_class::X_VERT_PLANE) {
79 p1.x = 0;
80 p2.x = 0;
81 point_diff = pcl::euclideanDistance(p1, p2);
82 }
83 if(plane_type == PlaneUtils::plane_class::Y_VERT_PLANE) {
84 p1.y = 0;
85 p2.y = 0;
86 point_diff = pcl::euclideanDistance(p1, p2);
87 }
88
89 return point_diff;

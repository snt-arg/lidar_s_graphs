36class EdgePlaneParallel : public BaseBinaryEdge<1, Eigen::Vector3d, VertexPlane, VertexPlane> {
37public:
38 EIGEN_MAKE_ALIGNED_OPERATOR_NEW
39 EdgePlaneParallel() : BaseBinaryEdge<1, Eigen::Vector3d, VertexPlane, VertexPlane>() {
40 _information.setIdentity();
41 _error.setZero();
42 }
43
44 void computeError() override {
45 const VertexPlane* v1 = static_cast<const VertexPlane*>(_vertices[0]);
46 const VertexPlane* v2 = static_cast<const VertexPlane*>(_vertices[1]);
47
48 double num = v1->estimate().normal().dot(v2->estimate().normal());
49 double den = v1->estimate().normal().norm() * (v2->estimate().normal().norm());
50
51 _error[0] = acos(fabs(num)/den);
52 }
53 virtual bool read(std::istream& is) override {
54 Eigen::Vector3d v;
55 for(int i = 0; i < 3; ++i) {
56 is >> v[i];
57 }
58
59 setMeasurement(v);
60 for(int i = 0; i < information().rows(); ++i) {
61 for(int j = i; j < information().cols(); ++j) {
62 is >> information()(i, j);
63 if(i != j) {
64 information()(j, i) = information()(i, j);
65 }
66 }
67 }
68
69 return true;
70 }
71
72 virtual bool write(std::ostream& os) const override {
73 for(int i = 0; i < 3; ++i) {
74 os << _measurement[i] << " ";
75 }
76
77 for(int i = 0; i < information().rows(); ++i) {
78 for(int j = i; j < information().cols(); ++j) {
79 os << " " << information()(i, j);
80 };
81 }
82 return os.good();
83 }
84
85 virtual void setMeasurement(const Eigen::Vector3d& m) override {
86 _measurement = m;
87 }

89 virtual int measurementDimension() const override {
90 return 3;
91 }
92};

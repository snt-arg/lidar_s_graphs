25static geometry_msgs::TransformStamped matrix2transform(const ros::Time& stamp, const Eigen::Matrix4f& pose, const std::string& frame_id, const std::string& child_frame_id) {
26 Eigen::Quaternionf quat(pose.block<3, 3>(0, 0));
27 quat.normalize();
28 geometry_msgs::Quaternion odom_quat;
29 odom_quat.w = quat.w();
30 odom_quat.x = quat.x();
31 odom_quat.y = quat.y();
32 odom_quat.z = quat.z();
33
34 geometry_msgs::TransformStamped odom_trans;
35 odom_trans.header.stamp = stamp;
36 odom_trans.header.frame_id = frame_id;
37 odom_trans.child_frame_id = child_frame_id;
38
39 odom_trans.transform.translation.x = pose(0, 3);
40 odom_trans.transform.translation.y = pose(1, 3);
41 odom_trans.transform.translation.z = pose(2, 3);
42 odom_trans.transform.rotation = odom_quat;
43
44 return odom_trans;
45}

77static Eigen::Isometry3d pose2isometry(const geometry_msgs::Pose& pose) {
78 Eigen::Isometry3d mat = Eigen::Isometry3d::Identity();
79 mat.translation() = Eigen::Vector3d(pose.position.x, pose.position.y, pose.position.z);
80 mat.linear() = Eigen::Quaterniond(pose.orientation.w, pose.orientation.x, pose.orientation.y, pose.orientation.z).toRotationMatrix();

125static Eigen::Isometry3d odom2isometry(const nav_msgs::OdometryConstPtr& odom_msg) {
126 const auto& orientation = odom_msg->pose.pose.orientation;
127 const auto& position = odom_msg->pose.pose.position;
128
129 Eigen::Quaterniond quat;
130 quat.w() = orientation.w;
131 quat.x() = orientation.x;
132 quat.y() = orientation.y;
133 quat.z() = orientation.z;

135 Eigen::Isometry3d isometry = Eigen::Isometry3d::Identity();
136 isometry.linear() = quat.toRotationMatrix();

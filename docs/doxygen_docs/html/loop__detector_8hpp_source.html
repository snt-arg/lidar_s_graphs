99 std::vector<KeyFrame::Ptr> find_candidates(const std::vector<KeyFrame::Ptr>& keyframes, const KeyFrame::Ptr& new_keyframe) const {
100 // too close to the last registered loop edge
101 if(new_keyframe->accum_distance - last_edge_accum_distance < distance_from_last_edge_thresh) {
102 return std::vector<KeyFrame::Ptr>();
103 }
104
105 std::vector<KeyFrame::Ptr> candidates;
106 candidates.reserve(32);
107
108 for(const auto& k : keyframes) {
109 // traveled distance between keyframes is too small
110 if(new_keyframe->accum_distance - k->accum_distance < accum_distance_thresh) {
111 continue;
112 }
113
114 const auto& pos1 = k->node->estimate().translation();
115 const auto& pos2 = new_keyframe->node->estimate().translation();
116
117 // estimated distance between keyframes is too small
118 double dist = (pos1.head<2>() - pos2.head<2>()).norm();
119 if(dist > distance_thresh) {
120 continue;
121 }
122
123 candidates.push_back(k);
124 }
125
126 return candidates;

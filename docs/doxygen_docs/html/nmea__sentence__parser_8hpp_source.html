113 int checksum_loc = sentence.find('*');
114 if(checksum_loc == std::string::npos) {
115 return GPRMC();
116 }
117
118 int checksum = std::stoul(sentence.substr(checksum_loc + 1), nullptr, 16);
119
120 std::string substr = sentence.substr(1, checksum_loc - 1);
121 int sum = std::accumulate(substr.begin(), substr.end(), static_cast<unsigned char>(0), [=](unsigned char n, unsigned char c) { return n ^ c; });
122
123 if(checksum != (sum & 0xf)) {
124 std::cerr << "checksum doesn't match!!" << std::endl;
125 std::cerr << sentence << " " << sum << std::endl;
126 return GPRMC();
127 }
128
129 std::vector<std::string> tokens;
130 boost::split(tokens, sentence, boost::is_any_of(","));

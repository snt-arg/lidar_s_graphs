77 inline float width_between_planes(Eigen::Vector4d v1, Eigen::Vector4d v2) {
78 Eigen::Vector3d vec;
79 float size = 0;
80
81 if(fabs(v1(3)) > fabs(v2(3)))
82 vec = fabs(v1(3)) * v1.head(3) - fabs(v2(3)) * v2.head(3);
83 else if(fabs(v2(3)) > fabs(v1(3)))
84 vec = fabs(v2(3)) * v2.head(3) - fabs(v1(3)) * v1.head(3);
85
86 size = fabs(vec(0) + vec(1));
87
88 return size;
89 }

152 Eigen::Vector2d room_center(const Eigen::Vector4d& x_plane1, const Eigen::Vector4d& x_plane2, const Eigen::Vector4d& y_plane1, const Eigen::Vector4d& y_plane2) {
153 Eigen::Vector2d center;
154 Eigen::Vector3d vec_x, vec_y;
155
156 if(fabs(x_plane1(3)) > fabs(x_plane2(3))) {
157 vec_x = (0.5 * (fabs(x_plane1(3)) * x_plane1.head(3) - fabs(x_plane2(3)) * x_plane2.head(3))) + fabs(x_plane2(3)) * x_plane2.head(3);
158 } else {
159 vec_x = (0.5 * (fabs(x_plane2(3)) * x_plane2.head(3) - fabs(x_plane1(3)) * x_plane1.head(3))) + fabs(x_plane1(3)) * x_plane1.head(3);
160 }
161
162 if(fabs(y_plane1(3)) > fabs(y_plane2(3))) {
163 vec_y = (0.5 * (fabs(y_plane1(3)) * y_plane1.head(3) - fabs(y_plane2(3)) * y_plane2.head(3))) + fabs(y_plane2(3)) * y_plane2.head(3);
164 } else {
165 vec_y = (0.5 * (fabs(y_plane2(3)) * y_plane2.head(3) - fabs(y_plane1(3)) * y_plane1.head(3))) + fabs(y_plane1(3)) * y_plane1.head(3);
166 }
167
168 Eigen::Vector3d final_vec = vec_x + vec_y;
169 center(0) = final_vec(0);
170 center(1) = final_vec(1);
171
172 return center;
173 }

260 pcl::PointXY convert_point_to_map(pcl::PointXY point_local, Eigen::Matrix4d keyframe_pose) {
261 pcl::PointXY point_map;
262
263 Eigen::Vector4d point_map_eigen, point_local_eigen;
264 point_local_eigen = point_map_eigen.setZero();
265 point_local_eigen(3) = point_map_eigen(3) = 1;
266 point_local_eigen(0) = point_local.x;
267 point_local_eigen(1) = point_local.y;
268 point_map_eigen = keyframe_pose * point_local_eigen;
269
270 point_map.x = point_map_eigen(0);
271 point_map.y = point_map_eigen(1);
272 return point_map;
273 }

281 float get_min_segment(const pcl::PointCloud<PointNormal>::Ptr& cloud_1, const pcl::PointCloud<PointNormal>::Ptr& cloud_2) {
282 float min_dist = std::numeric_limits<float>::max();
283 const auto token = std::numeric_limits<std::size_t>::max();
284 std::size_t i_min = token, i_max = token;
285
286 for(std::size_t i = 0; i < cloud_1->points.size(); ++i) {
287 for(std::size_t j = 0; j < cloud_2->points.size(); ++j) {
288 // Compute the distance
289 float dist = (cloud_1->points[i].getVector4fMap() - cloud_2->points[j].getVector4fMap()).squaredNorm();
290 if(dist >= min_dist) continue;
291
292 min_dist = dist;
293 i_min = i;
294 i_max = j;
295 }
296 }
297
298 return (std::sqrt(min_dist));
299 }

307 bool check_point_neighbours(const pcl::PointCloud<PointNormal>::Ptr& cloud_1, const pcl::PointCloud<PointNormal>::Ptr& cloud_2) {
308 bool valid_neighbour = false;
309 int point_count = 0;
310 float min_dist = 0.5;
311
312 for(std::size_t i = 0; i < cloud_1->points.size(); ++i) {
313 for(std::size_t j = 0; j < cloud_2->points.size(); ++j) {
314 // Compute the distance
315 float dist = (cloud_1->points[i].getVector4fMap() - cloud_2->points[j].getVector4fMap()).squaredNorm();
316 if(dist < min_dist) {
317 point_count++;
318 break;
319 }
320 }
321 if(point_count > 100) {
322 valid_neighbour = true;
323 break;
324 }
325 }
326 return valid_neighbour;
327 }

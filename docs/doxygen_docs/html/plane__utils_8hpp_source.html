62

64  vec = fabs(v1(3)) * v1.head(3) - fabs(v2(3)) * v2.head(3);
65  else if(fabs(v2(3)) > fabs(v1(3)))
66  vec = fabs(v2(3)) * v2.head(3) - fabs(v1(3)) * v1.head(3);

209  float get_min_segment(const pcl::PointCloud<PointNormal>::Ptr& cloud_1, const pcl::PointCloud<PointNormal>::Ptr& cloud_2) {
210  float min_dist = std::numeric_limits<float>::max();
211  const auto token = std::numeric_limits<std::size_t>::max();
212  std::size_t i_min = token, i_max = token;
213 
214  for(std::size_t i = 0; i < cloud_1->points.size(); ++i) {
215  for(std::size_t j = 0; j < cloud_2->points.size(); ++j) {
216  // Compute the distance
217  float dist = (cloud_1->points[i].getVector4fMap() - cloud_2->points[j].getVector4fMap()).squaredNorm();
218  if(dist >= min_dist) continue;
219 
220  min_dist = dist;
221  i_min = i;
222  i_max = j;
223  }
224  }
225 
226  return (std::sqrt(min_dist));
227  }

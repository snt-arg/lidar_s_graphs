34 virtual void onInit() {
35 nh = getNodeHandle();
36 private_nh = getPrivateNodeHandle();
37
38 initialize_params();
39
40 if(private_nh.param<bool>("deskewing", false)) {
41 imu_sub = nh.subscribe("/imu/data", 1, &PrefilteringNodelet::imu_callback, this);

43
44 points_sub = nh.subscribe("/velodyne_points", 64, &PrefilteringNodelet::cloud_callback, this);
45 points_pub = nh.advertise<sensor_msgs::PointCloud2>("/filtered_points", 32);

49private:
50 void initialize_params() {
51 std::string downsample_method = private_nh.param<std::string>("downsample_method", "VOXELGRID");
52 double downsample_resolution = private_nh.param<double>("downsample_resolution", 0.1);
53
54 if(downsample_method == "VOXELGRID") {
55 std::cout << "downsample: VOXELGRID " << downsample_resolution << std::endl;
56 boost::shared_ptr<pcl::VoxelGrid<PointT>> voxelgrid(new pcl::VoxelGrid<PointT>());
57 voxelgrid->setLeafSize(downsample_resolution, downsample_resolution, downsample_resolution);
58 downsample_filter = voxelgrid;
59 } else if(downsample_method == "APPROX_VOXELGRID") {
60 std::cout << "downsample: APPROX_VOXELGRID " << downsample_resolution << std::endl;
61 boost::shared_ptr<pcl::ApproximateVoxelGrid<PointT>> approx_voxelgrid(new pcl::ApproximateVoxelGrid<PointT>());
62 approx_voxelgrid->setLeafSize(downsample_resolution, downsample_resolution, downsample_resolution);
63 downsample_filter = approx_voxelgrid;
64 } else {
65 if(downsample_method != "NONE") {
66 std::cerr << "warning: unknown downsampling type (" << downsample_method << ")" << std::endl;
67 std::cerr << " : use passthrough filter" << std::endl;
68 }
69 std::cout << "downsample: NONE" << std::endl;
70 }
71
72 std::string outlier_removal_method = private_nh.param<std::string>("outlier_removal_method", "STATISTICAL");
73 if(outlier_removal_method == "STATISTICAL") {
74 int mean_k = private_nh.param<int>("statistical_mean_k", 20);
75 double stddev_mul_thresh = private_nh.param<double>("statistical_stddev", 1.0);
76 std::cout << "outlier_removal: STATISTICAL " << mean_k << " - " << stddev_mul_thresh << std::endl;
77
78 pcl::StatisticalOutlierRemoval<PointT>::Ptr sor(new pcl::StatisticalOutlierRemoval<PointT>());
79 sor->setMeanK(mean_k);
80 sor->setStddevMulThresh(stddev_mul_thresh);
81 outlier_removal_filter = sor;
82 } else if(outlier_removal_method == "RADIUS") {
83 double radius = private_nh.param<double>("radius_radius", 0.8);
84 int min_neighbors = private_nh.param<int>("radius_min_neighbors", 2);
85 std::cout << "outlier_removal: RADIUS " << radius << " - " << min_neighbors << std::endl;
86
87 pcl::RadiusOutlierRemoval<PointT>::Ptr rad(new pcl::RadiusOutlierRemoval<PointT>());
88 rad->setRadiusSearch(radius);
89 rad->setMinNeighborsInRadius(min_neighbors);
90 outlier_removal_filter = rad;
91 } else {
92 std::cout << "outlier_removal: NONE" << std::endl;
93 }
94
95 use_distance_filter = private_nh.param<bool>("use_distance_filter", true);
96 distance_near_thresh = private_nh.param<double>("distance_near_thresh", 1.0);
97 distance_far_thresh = private_nh.param<double>("distance_far_thresh", 100.0);
98
99 base_link_frame = private_nh.param<std::string>("base_link_frame", "");
100 }

102 void imu_callback(const sensor_msgs::ImuConstPtr& imu_msg) {
103 imu_queue.push_back(imu_msg);
104 }

106 void cloud_callback(pcl::PointCloud<PointT>::ConstPtr src_cloud) {
107 if(src_cloud->empty()) {
108 return;
109 }
110
111 src_cloud = deskewing(src_cloud);
112
113 // if base_link_frame is defined, transform the input cloud to the frame
114 if(!base_link_frame.empty()) {
115 if(!tf_listener.canTransform(base_link_frame, src_cloud->header.frame_id, ros::Time(0))) {
116 std::cerr << "failed to find transform between " << base_link_frame << " and " << src_cloud->header.frame_id << std::endl;
117 }
118
119 tf::StampedTransform transform;
120 tf_listener.waitForTransform(base_link_frame, src_cloud->header.frame_id, ros::Time(0), ros::Duration(2.0));
121 tf_listener.lookupTransform(base_link_frame, src_cloud->header.frame_id, ros::Time(0), transform);
122
123 pcl::PointCloud<PointT>::Ptr transformed(new pcl::PointCloud<PointT>());
124 pcl_ros::transformPointCloud(*src_cloud, *transformed, transform);
125 transformed->header.frame_id = base_link_frame;
126 transformed->header.stamp = src_cloud->header.stamp;
127 src_cloud = transformed;
128 }
129
130 pcl::PointCloud<PointT>::ConstPtr filtered = distance_filter(src_cloud);
131 filtered = downsample(filtered);
132 filtered = outlier_removal(filtered);
133
134 points_pub.publish(filtered);
135 }

137 pcl::PointCloud<PointT>::ConstPtr downsample(const pcl::PointCloud<PointT>::ConstPtr& cloud) const {
138 if(!downsample_filter) {
139 return cloud;
140 }
141
142 pcl::PointCloud<PointT>::Ptr filtered(new pcl::PointCloud<PointT>());
143 downsample_filter->setInputCloud(cloud);
144 downsample_filter->filter(*filtered);
145 filtered->header = cloud->header;
146
147 return filtered;
148 }
149
150 pcl::PointCloud<PointT>::ConstPtr outlier_removal(const pcl::PointCloud<PointT>::ConstPtr& cloud) const {
151 if(!outlier_removal_filter) {
152 return cloud;
153 }
154
155 pcl::PointCloud<PointT>::Ptr filtered(new pcl::PointCloud<PointT>());
156 outlier_removal_filter->setInputCloud(cloud);
157 outlier_removal_filter->filter(*filtered);
158 filtered->header = cloud->header;
159
160 return filtered;
161 }

163 pcl::PointCloud<PointT>::ConstPtr distance_filter(const pcl::PointCloud<PointT>::ConstPtr& cloud) const {
164 pcl::PointCloud<PointT>::Ptr filtered(new pcl::PointCloud<PointT>());
165 filtered->reserve(cloud->size());
166
167 std::copy_if(cloud->begin(), cloud->end(), std::back_inserter(filtered->points), [&](const PointT& p) {
168 double d = p.getVector3fMap().norm();
169 return d > distance_near_thresh && d < distance_far_thresh;
170 });
171
172 filtered->width = filtered->size();
173 filtered->height = 1;
174 filtered->is_dense = false;
175
176 filtered->header = cloud->header;
177
178 return filtered;
179 }
180
181 pcl::PointCloud<PointT>::ConstPtr deskewing(const pcl::PointCloud<PointT>::ConstPtr& cloud) {
182 ros::Time stamp = pcl_conversions::fromPCL(cloud->header.stamp);
183 if(imu_queue.empty()) {
184 return cloud;
185 }
186
187 // the color encodes the point number in the point sequence
188 if(colored_pub.getNumSubscribers()) {
189 pcl::PointCloud<pcl::PointXYZRGB>::Ptr colored(new pcl::PointCloud<pcl::PointXYZRGB>());
190 colored->header = cloud->header;
191 colored->is_dense = cloud->is_dense;
192 colored->width = cloud->width;
193 colored->height = cloud->height;
194 colored->resize(cloud->size());
195
196 for(int i = 0; i < cloud->size(); i++) {
197 double t = static_cast<double>(i) / cloud->size();
198 colored->at(i).getVector4fMap() = cloud->at(i).getVector4fMap();
199 colored->at(i).r = 255 * t;
200 colored->at(i).g = 128;
201 colored->at(i).b = 255 * (1 - t);
202 }
203 colored_pub.publish(colored);
204 }
205
206 sensor_msgs::ImuConstPtr imu_msg = imu_queue.front();
207
208 auto loc = imu_queue.begin();
209 for(; loc != imu_queue.end(); loc++) {
210 imu_msg = (*loc);
211 if((*loc)->header.stamp > stamp) {
212 break;
213 }
214 }
215
216 imu_queue.erase(imu_queue.begin(), loc);
217
218 Eigen::Vector3f ang_v(imu_msg->angular_velocity.x, imu_msg->angular_velocity.y, imu_msg->angular_velocity.z);
219 ang_v *= -1;
220
221 pcl::PointCloud<PointT>::Ptr deskewed(new pcl::PointCloud<PointT>());
222 deskewed->header = cloud->header;
223 deskewed->is_dense = cloud->is_dense;
224 deskewed->width = cloud->width;
225 deskewed->height = cloud->height;
226 deskewed->resize(cloud->size());
227
228 double scan_period = private_nh.param<double>("scan_period", 0.1);
229 for(int i = 0; i < cloud->size(); i++) {
230 const auto& pt = cloud->at(i);
231
232 // TODO: transform IMU data into the LIDAR frame
233 double delta_t = scan_period * static_cast<double>(i) / cloud->size();
234 Eigen::Quaternionf delta_q(1, delta_t / 2.0 * ang_v[0], delta_t / 2.0 * ang_v[1], delta_t / 2.0 * ang_v[2]);
235 Eigen::Vector3f pt_ = delta_q.inverse() * pt.getVector3fMap();
236
237 deskewed->at(i) = cloud->at(i);
238 deskewed->at(i).getVector3fMap() = pt_;
239 }
240
241 return deskewed;
242 }

96  } else {

98  std::cerr << "warning: unknown downsampling type (" << downsample_method << ")" << std::endl;
99  std::cerr << " : use passthrough filter" << std::endl;

165  Eigen::Matrix4f matching(const ros::Time& stamp, const pcl::PointCloud<PointT>::ConstPtr& cloud) {
166  if(!keyframe) {
167  prev_time = ros::Time();
168  prev_trans.setIdentity();
169  keyframe_pose.setIdentity();
170  keyframe_stamp = stamp;
171  keyframe = downsample(cloud);
172  registration->setInputTarget(keyframe);
173  return Eigen::Matrix4f::Identity();
174  }
175 
176  auto filtered = downsample(cloud);
177  registration->setInputSource(filtered);
178 
179  std::string msf_source;
180  Eigen::Isometry3f msf_delta = Eigen::Isometry3f::Identity();
181 
182  if(private_nh.param<bool>("enable_imu_frontend", false)) {
183  if(msf_pose && msf_pose->header.stamp > keyframe_stamp && msf_pose_after_update && msf_pose_after_update->header.stamp > keyframe_stamp) {
184  Eigen::Isometry3d pose0 = pose2isometry(msf_pose_after_update->pose.pose);
185  Eigen::Isometry3d pose1 = pose2isometry(msf_pose->pose.pose);
186  Eigen::Isometry3d delta = pose0.inverse() * pose1;
187 
188  msf_source = "imu";
189  msf_delta = delta.cast<float>();
190  } else {
191  std::cerr << "msf data is too old" << std::endl;
192  }
193  } else if(private_nh.param<bool>("enable_robot_odometry_init_guess", false) && !prev_time.isZero()) {
194  tf::StampedTransform transform;
195  if(tf_listener.waitForTransform(cloud->header.frame_id, stamp, cloud->header.frame_id, prev_time, robot_odom_frame_id, ros::Duration(0))) {
196  tf_listener.lookupTransform(cloud->header.frame_id, stamp, cloud->header.frame_id, prev_time, robot_odom_frame_id, transform);
197  } else if(tf_listener.waitForTransform(cloud->header.frame_id, ros::Time(0), cloud->header.frame_id, prev_time, robot_odom_frame_id, ros::Duration(0))) {
198  tf_listener.lookupTransform(cloud->header.frame_id, ros::Time(0), cloud->header.frame_id, prev_time, robot_odom_frame_id, transform);
199  }
200 
201  if(transform.stamp_.isZero()) {
202  NODELET_WARN_STREAM("failed to look up transform between " << cloud->header.frame_id << " and " << robot_odom_frame_id);
203  } else {
204  msf_source = "odometry";
205  msf_delta = tf2isometry(transform).cast<float>();
206  }
207  }
208 
209  pcl::PointCloud<PointT>::Ptr aligned(new pcl::PointCloud<PointT>());
210  registration->align(*aligned, prev_trans * msf_delta.matrix());
211 
212  publish_scan_matching_status(stamp, cloud->header.frame_id, aligned, msf_source, msf_delta);
213 
214  if(!registration->hasConverged()) {
215  NODELET_INFO_STREAM("scan matching has not converged!!");
216  NODELET_INFO_STREAM("ignore this frame(" << stamp << ")");
217  return keyframe_pose * prev_trans;
218  }
219 
220  Eigen::Matrix4f trans = registration->getFinalTransformation();
221  Eigen::Matrix4f odom = keyframe_pose * trans;
222 
223  if(transform_thresholding) {
224  Eigen::Matrix4f delta = prev_trans.inverse() * trans;
225  double dx = delta.block<3, 1>(0, 3).norm();
226  double da = std::acos(Eigen::Quaternionf(delta.block<3, 3>(0, 0)).w());
227 
228  if(dx > max_acceptable_trans || da > max_acceptable_angle) {
229  NODELET_INFO_STREAM("too large transform!! " << dx << "[m] " << da << "[rad]");
230  NODELET_INFO_STREAM("ignore this frame(" << stamp << ")");
231  return keyframe_pose * prev_trans;
232  }
233  }
234 
235  prev_time = stamp;
236  prev_trans = trans;
237 
238  auto keyframe_trans = matrix2transform(stamp, keyframe_pose, odom_frame_id, "keyframe");
239  keyframe_broadcaster.sendTransform(keyframe_trans);
240 
241  double delta_trans = trans.block<3, 1>(0, 3).norm();
242  double delta_angle = std::acos(Eigen::Quaternionf(trans.block<3, 3>(0, 0)).w());
243  double delta_time = (stamp - keyframe_stamp).toSec();
244  if(delta_trans > keyframe_delta_trans || delta_angle > keyframe_delta_angle || delta_time > keyframe_delta_time) {
245  keyframe = filtered;
246  registration->setInputTarget(keyframe);
247 
248  keyframe_pose = odom;
249  keyframe_stamp = stamp;
250  prev_time = stamp;
251  prev_trans.setIdentity();
252  }
253 
254  if(aligned_points_pub.getNumSubscribers() > 0) {
255  pcl::transformPointCloud(*cloud, *aligned, odom);
256  aligned->header.frame_id = odom_frame_id;
257  aligned_points_pub.publish(aligned);
258  }
259 
260  return odom;
261  }

268  void publish_odometry(const ros::Time& stamp, const std::string& base_frame_id, const Eigen::Matrix4f& pose) {
269  // publish transform stamped for IMU integration
270  geometry_msgs::TransformStamped odom_trans = matrix2transform(stamp, pose, odom_frame_id, base_frame_id);
271  trans_pub.publish(odom_trans);
272 
273  // broadcast the transform over tf
274  if(publish_tf) odom_broadcaster.sendTransform(odom_trans);
275 
276  // publish the transform
277  nav_msgs::Odometry odom;
278  odom.header.stamp = stamp;
279  odom.header.frame_id = odom_frame_id;
280 
281  odom.pose.pose.position.x = pose(0, 3);
282  odom.pose.pose.position.y = pose(1, 3);
283  odom.pose.pose.position.z = pose(2, 3);
284  odom.pose.pose.orientation = odom_trans.transform.rotation;
285 
286  odom.child_frame_id = base_frame_id;
287  odom.twist.twist.linear.x = 0.0;
288  odom.twist.twist.linear.y = 0.0;
289  odom.twist.twist.angular.z = 0.0;
290 
291  odom_pub.publish(odom);
292  }

297  void publish_scan_matching_status(const ros::Time& stamp, const std::string& frame_id, pcl::PointCloud<pcl::PointXYZI>::ConstPtr aligned, const std::string& msf_source, const Eigen::Isometry3f& msf_delta) {
298  if(!status_pub.getNumSubscribers()) {
299  return;
300  }
301 
302  ScanMatchingStatus status;
303  status.header.frame_id = frame_id;
304  status.header.stamp = stamp;
305  status.has_converged = registration->hasConverged();
306  status.matching_error = registration->getFitnessScore();
307 
308  const double max_correspondence_dist = 0.5;
309 
310  int num_inliers = 0;
311  std::vector<int> k_indices;
312  std::vector<float> k_sq_dists;
313  for(int i = 0; i < aligned->size(); i++) {
314  const auto& pt = aligned->at(i);
315  registration->getSearchMethodTarget()->nearestKSearch(pt, 1, k_indices, k_sq_dists);
316  if(k_sq_dists[0] < max_correspondence_dist * max_correspondence_dist) {
317  num_inliers++;
318  }
319  }
320  status.inlier_fraction = static_cast<float>(num_inliers) / aligned->size();
321 
322  status.relative_pose = isometry2pose(Eigen::Isometry3f(registration->getFinalTransformation()).cast<double>());
323 
324  if(!msf_source.empty()) {
325  status.prediction_labels.resize(1);
326  status.prediction_labels[0].data = msf_source;
327 
328  status.prediction_errors.resize(1);
329  Eigen::Isometry3f error = Eigen::Isometry3f(registration->getFinalTransformation()).inverse() * msf_delta;
330  status.prediction_errors[0] = isometry2pose(error.cast<double>());
331  }
332 
333  status_pub.publish(status);
334  }

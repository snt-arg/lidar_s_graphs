54  void initialize_params() {
55  tilt_deg = private_nh.param<double>("tilt_deg", 0.0); // approximate sensor tilt angle [deg]
56  sensor_height = private_nh.param<double>("sensor_height", 2.0); // approximate sensor height [m]
57  height_clip_range = private_nh.param<double>("height_clip_range", 1.0); // points with heights in [sensor_height - height_clip_range, sensor_height + height_clip_range] will be used for floor detection
58  floor_pts_thresh = private_nh.param<int>("floor_pts_thresh", 512); // minimum number of support points of RANSAC to accept a detected floor plane
59  floor_normal_thresh = private_nh.param<double>("floor_normal_thresh", 10.0); // verticality check thresold for the detected floor plane [deg]
60  use_normal_filtering = private_nh.param<bool>("use_normal_filtering", true); // if true, points with "non-"vertical normals will be filtered before RANSAC
61  normal_filter_thresh = private_nh.param<double>("normal_filter_thresh", 20.0); // "non-"verticality check threshold [deg]
62 
63  points_topic = private_nh.param<std::string>("points_topic", "/velodyne_points");

70  void cloud_callback(const sensor_msgs::PointCloud2ConstPtr& cloud_msg) {
71  pcl::PointCloud<PointT>::Ptr cloud(new pcl::PointCloud<PointT>());
72  pcl::fromROSMsg(*cloud_msg, *cloud);
73 
74  if(cloud->empty()) {
75  return;
76  }
77 
78  // floor detection
79  boost::optional<Eigen::Vector4f> floor = detect(cloud);
80

84  if(floor) {
85  coeffs.coeffs.resize(4);
86  for(int i = 0; i < 4; i++) {

89  }

91  floor_pub.publish(coeffs);
92 
93  // for offline estimation

95  read_until->frame_id = points_topic;

97  read_until_pub.publish(read_until);
98 
99  read_until->frame_id = "/filtered_points";
100  read_until_pub.publish(read_until);

108  boost::optional<Eigen::Vector4f> detect(const pcl::PointCloud<PointT>::Ptr& cloud) const {
109  // compensate the tilt rotation
110  Eigen::Matrix4f tilt_matrix = Eigen::Matrix4f::Identity();
111  tilt_matrix.topLeftCorner(3, 3) = Eigen::AngleAxisf(tilt_deg * M_PI / 180.0f, Eigen::Vector3f::UnitY()).toRotationMatrix();
112 
113  // filtering before RANSAC (height and normal filtering)
114  pcl::PointCloud<PointT>::Ptr filtered(new pcl::PointCloud<PointT>);
115  pcl::transformPointCloud(*cloud, *filtered, tilt_matrix);
116  filtered = plane_clip(filtered, Eigen::Vector4f(0.0f, 0.0f, 1.0f, sensor_height + height_clip_range), false);
117  filtered = plane_clip(filtered, Eigen::Vector4f(0.0f, 0.0f, 1.0f, sensor_height - height_clip_range), true);
118 
119  if(use_normal_filtering) {
120  filtered = normal_filtering(filtered);
121  }
122 
123  pcl::transformPointCloud(*filtered, *filtered, static_cast<Eigen::Matrix4f>(tilt_matrix.inverse()));
124 
125  if(floor_filtered_pub.getNumSubscribers()) {
126  filtered->header = cloud->header;
127  floor_filtered_pub.publish(filtered);
128  }
129 
130  // too few points for RANSAC
131  if(filtered->size() < floor_pts_thresh) {
132  return boost::none;
133  }
134 
135  // RANSAC
136  pcl::SampleConsensusModelPlane<PointT>::Ptr model_p(new pcl::SampleConsensusModelPlane<PointT>(filtered));
137  pcl::RandomSampleConsensus<PointT> ransac(model_p);
138  ransac.setDistanceThreshold(0.1);
139  ransac.computeModel();
140 
141  pcl::PointIndices::Ptr inliers(new pcl::PointIndices);
142  ransac.getInliers(inliers->indices);
143 
144  // too few inliers
145  if(inliers->indices.size() < floor_pts_thresh) {
146  return boost::none;
147  }
148 
149  // verticality check of the detected floor's normal
150  Eigen::Vector4f reference = tilt_matrix.inverse() * Eigen::Vector4f::UnitZ();
151 
152  Eigen::VectorXf coeffs;
153  ransac.getModelCoefficients(coeffs);
154 
155  double dot = coeffs.head<3>().dot(reference.head<3>());
156  if(std::abs(dot) < std::cos(floor_normal_thresh * M_PI / 180.0)) {
157  // the normal is not vertical
158  return boost::none;
159  }
160 
161  // make the normal upward
162  if(coeffs.head<3>().dot(Eigen::Vector3f::UnitZ()) < 0.0f) {
163  coeffs *= -1.0f;
164  }
165 
166  if(floor_points_pub.getNumSubscribers()) {
167  pcl::PointCloud<PointT>::Ptr inlier_cloud(new pcl::PointCloud<PointT>);
168  pcl::ExtractIndices<PointT> extract;
169  extract.setInputCloud(filtered);
170  extract.setIndices(inliers);
171  extract.filter(*inlier_cloud);
172  inlier_cloud->header = cloud->header;
173 
174  floor_points_pub.publish(inlier_cloud);
175  }
176 
177  return Eigen::Vector4f(coeffs);
178  }

187  pcl::PointCloud<PointT>::Ptr plane_clip(const pcl::PointCloud<PointT>::Ptr& src_cloud, const Eigen::Vector4f& plane, bool negative) const {
188  pcl::PlaneClipper3D<PointT> clipper(plane);
189  pcl::PointIndices::Ptr indices(new pcl::PointIndices);
190 
191  clipper.clipPointCloud3D(*src_cloud, indices->indices);
192 
193  pcl::PointCloud<PointT>::Ptr dst_cloud(new pcl::PointCloud<PointT>);
194 
195  pcl::ExtractIndices<PointT> extract;
196  extract.setInputCloud(src_cloud);
197  extract.setIndices(indices);
198  extract.setNegative(negative);
199  extract.filter(*dst_cloud);
200 
201  return dst_cloud;
202  }

209  pcl::PointCloud<PointT>::Ptr normal_filtering(const pcl::PointCloud<PointT>::Ptr& cloud) const {
210  pcl::NormalEstimation<PointT, pcl::Normal> ne;
211  ne.setInputCloud(cloud);
212 
213  pcl::search::KdTree<PointT>::Ptr tree(new pcl::search::KdTree<PointT>);
214  ne.setSearchMethod(tree);
215 
216  pcl::PointCloud<pcl::Normal>::Ptr normals(new pcl::PointCloud<pcl::Normal>);
217  ne.setKSearch(10);
218  ne.setViewPoint(0.0f, 0.0f, sensor_height);
219  ne.compute(*normals);
220 
221  pcl::PointCloud<PointT>::Ptr filtered(new pcl::PointCloud<PointT>);
222  filtered->reserve(cloud->size());
223 
224  for(int i = 0; i < cloud->size(); i++) {
225  float dot = normals->at(i).getNormalVector3fMap().normalized().dot(Eigen::Vector3f::UnitZ());
226  if(std::abs(dot) > std::cos(normal_filter_thresh * M_PI / 180.0)) {
227  filtered->push_back(cloud->at(i));
228  }
229  }
230 
231  filtered->width = filtered->size();
232  filtered->height = 1;
233  filtered->is_dense = false;
234 
235  return filtered;
236  }

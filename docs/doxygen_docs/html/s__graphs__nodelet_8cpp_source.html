483  for(auto& keyframe : keyframes) {
484  if(keyframe->stamp > gps_queue.back()->header.stamp) {
485  break;
486  }
487 
488  if(keyframe->stamp < (*gps_cursor)->header.stamp || keyframe->utm_coord) {
489  continue;
490  }
491 
492  // find the gps data which is closest to the keyframe
493  auto closest_gps = gps_cursor;
494  for(auto gps = gps_cursor; gps != gps_queue.end(); gps++) {
495  auto dt = ((*closest_gps)->header.stamp - keyframe->stamp).toSec();
496  auto dt2 = ((*gps)->header.stamp - keyframe->stamp).toSec();
497  if(std::abs(dt) < std::abs(dt2)) {
498  break;
499  }
500 
501  closest_gps = gps;
502  }
503 
504  // if the time residual between the gps and keyframe is too large, skip it
505  gps_cursor = closest_gps;
506  if(0.2 < std::abs(((*closest_gps)->header.stamp - keyframe->stamp).toSec())) {
507  continue;
508  }
509 
510  // convert (latitude, longitude, altitude) -> (easting, northing, altitude) in UTM coordinate
511  geodesy::UTMPoint utm;
512  geodesy::fromMsg((*closest_gps)->position, utm);
513  Eigen::Vector3d xyz(utm.easting, utm.northing, utm.altitude);
514 
515  // the first gps data position will be the origin of the map
516  if(!zero_utm) {
517  zero_utm = xyz;
518  }
519  xyz -= (*zero_utm);
520 
521  keyframe->utm_coord = xyz;
522 
523  g2o::OptimizableGraph::Edge* edge;
524  if(std::isnan(xyz.z())) {
525  Eigen::Matrix2d information_matrix = Eigen::Matrix2d::Identity() / gps_edge_stddev_xy;
526  edge = graph_slam->add_se3_prior_xy_edge(keyframe->node, xyz.head<2>(), information_matrix);
527  } else {
528  Eigen::Matrix3d information_matrix = Eigen::Matrix3d::Identity();
529  information_matrix.block<2, 2>(0, 0) /= gps_edge_stddev_xy;
530  information_matrix(2, 2) /= gps_edge_stddev_z;
531  edge = graph_slam->add_se3_prior_xyz_edge(keyframe->node, xyz, information_matrix);
532  }
533  graph_slam->add_robust_kernel(edge, private_nh.param<std::string>("gps_edge_robust_kernel", "NONE"), private_nh.param<double>("gps_edge_robust_kernel_size", 1.0));
534 
535  updated = true;
536  }
537 
538  auto remove_loc = std::upper_bound(gps_queue.begin(), gps_queue.end(), keyframes.back()->stamp, [=](const ros::Time& stamp, const geographic_msgs::GeoPointStampedConstPtr& geopoint) { return stamp < geopoint->header.stamp; });
539  gps_queue.erase(gps_queue.begin(), remove_loc);
540  return updated;

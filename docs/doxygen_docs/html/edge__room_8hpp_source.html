228 class EdgeRoom2Planes : public BaseMultiEdge<2, Eigen::Vector2d> {
229 public:
230  EIGEN_MAKE_ALIGNED_OPERATOR_NEW
231  EdgeRoom2Planes() : BaseMultiEdge<2, Eigen::Vector2d>() {
232  resize(4);
233  }
234 
235  void computeError() override {
236  const VertexRoomXYLB* v1 = static_cast<const VertexRoomXYLB*>(_vertices[0]);
237  const VertexPlane* v2 = static_cast<const VertexPlane*>(_vertices[1]);
238  const VertexPlane* v3 = static_cast<const VertexPlane*>(_vertices[2]);
239  const VertexRoomXYLB* v4 = static_cast<const VertexRoomXYLB*>(_vertices[3]);
240 
241  Eigen::Vector2d room_pose = v1->estimate();
242  Eigen::Vector4d plane1 = v2->estimate().coeffs();
243  Eigen::Vector4d plane2 = v3->estimate().coeffs();
244  Eigen::Vector2d cluster_center = v4->estimate();
245 
246  correct_plane_d(plane1);
247  correct_plane_d(plane2);
248 
249  Eigen::Vector3d vec;
250  if(fabs(plane1(3)) > fabs(plane2(3))) {
251  vec = (0.5 * (fabs(plane1(3)) * plane1.head(3) - fabs(plane2(3)) * plane2.head(3))) + fabs(plane2(3)) * plane2.head(3);
252  } else {
253  vec = (0.5 * (fabs(plane2(3)) * plane2.head(3) - fabs(plane1(3)) * plane1.head(3))) + fabs(plane1(3)) * plane1.head(3);
254  }
255 
256  Eigen::Vector2d vec_normal = vec.head(2) / vec.norm();
257  Eigen::Vector2d final_pose_vec = vec.head(2) + (cluster_center - (cluster_center.dot(vec_normal)) * vec_normal);
258 
259  _error = room_pose - final_pose_vec;
260  }

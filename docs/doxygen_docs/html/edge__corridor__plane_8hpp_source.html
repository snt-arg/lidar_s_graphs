90 public:

94  void computeError() override {
95  const VertexCorridor* v1 = static_cast<const VertexCorridor*>(_vertices[0]);
96  const VertexPlane* v2 = static_cast<const VertexPlane*>(_vertices[1]);
97  double trans = v1->estimate();
98  Eigen::Vector4d plane = v2->estimate().coeffs();
99 
100  if(plane(3) > 0) {
101  plane(0) = -1 * plane(0);
102  plane(1) = -1 * plane(1);
103  plane(2) = -1 * plane(2);
104  plane(3) = -1 * plane(3);
105  }
106 
107  double est;
108  if(fabs(trans) > fabs(plane(3))) {
109  est = trans - plane(3);
110  } else {
111  est = plane(3) - trans;
112  }
113 
114  if(est * _measurement < 0) {
115  est = -1 * est;
116  }
117 
118  _error[0] = est - _measurement;
119  }

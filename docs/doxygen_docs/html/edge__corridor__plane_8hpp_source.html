43 void computeError() override {
44 const VertexSE3* v1 = static_cast<const VertexSE3*>(_vertices[0]);
45 const VertexCorridor* v2 = static_cast<const VertexCorridor*>(_vertices[1]);
46 Eigen::Isometry3d w2m = v1->estimate().inverse();
47 Eigen::Isometry3d corridor_pose_map;
48 corridor_pose_map.matrix().block<4, 4>(0, 0) = Eigen::Matrix4d::Identity();
49 corridor_pose_map.matrix()(1, 3) = v2->estimate();
50
51 Eigen::Isometry3d corridor_pose_local = corridor_pose_map * w2m;
52 double est = corridor_pose_local.matrix()(0, 3);
53
54 _error[0] = est - _measurement;
55 }

89class EdgeCorridorXPlane : public BaseBinaryEdge<1, double, g2o::VertexCorridor, g2o::VertexPlane> {
90public:
91 EIGEN_MAKE_ALIGNED_OPERATOR_NEW
92 EdgeCorridorXPlane() : BaseBinaryEdge<1, double, g2o::VertexCorridor, g2o::VertexPlane>() {}
93
94 void computeError() override {
95 const VertexCorridor* v1 = static_cast<const VertexCorridor*>(_vertices[0]);
96 const VertexPlane* v2 = static_cast<const VertexPlane*>(_vertices[1]);
97 double trans = v1->estimate();
98 Eigen::Vector4d plane = v2->estimate().coeffs();
99
100 if(plane(3) > 0) {
101 plane(0) = -1 * plane(0);
102 plane(1) = -1 * plane(1);
103 plane(2) = -1 * plane(2);
104 plane(3) = -1 * plane(3);
105 }
106
107 double est;
108 if(fabs(trans) > fabs(plane(3))) {
109 est = trans - plane(3);
110 } else {
111 est = plane(3) - trans;
112 }
113
114 if(est * _measurement < 0) {
115 est = -1 * est;
116 }
117
118 _error[0] = est - _measurement;
119 }

40 void computeError() override {
41 const g2o::VertexSE3* v1 = static_cast<const g2o::VertexSE3*>(_vertices[0]);
42 const g2o::VertexPlane* v2 = static_cast<const g2o::VertexPlane*>(_vertices[1]);
43
44 Eigen::Matrix4d Ti = v1->estimate().matrix();
45 Eigen::Vector4d Pj = v2->estimate().toVector();
46 Eigen::Matrix4d Gij = _measurement;
47 _error = Pj.transpose() * Ti * Gij * Ti.transpose() * Pj / 2;
48 }
